تعريف الدوال

الدوال (Functions) هي كتل من التعليمات البرمجية التي تؤدي مهمة محددة ويمكن استدعاؤها عند الحاجة. 


void main() {
  // تعريف دالة بسيطة
  void greet() {
    print('مرحباً بكم!');
  }
  
  greet(); // استدعاء الدالة
  
  // دالة مع قيمة إرجاع
  int add(int a, int b) {
    return a + b;
  }
  
  var result = add(5, 3);
  print(result); // 8
}

أنواع الدوال

النوع الوصف مثال
دوال عادية دوال محددة باسم ويمكن استدعاؤها مباشرة int sum(int a, int b) { return a + b; }
دوال مجهولة (Anonymous) دوال بدون اسم، تُستخدم كقيمة لمتغير var square = (int x) { return x * x; };
دوال سهمية (Arrow) صيغة مختصرة للدوال ذات التعبير الواحد var cube = (int x) => x * x * x;
دوال عليا (Higher-order) دوال تقبل دوال أخرى كمعاملات أو تُرجع دوالاً void forEach(void action(int)) { ... }

معاملات الدوال

ملاحظة هامة

في Dart لدينا نوعان رئيسيان من معاملات الدوال: Positional و Named. 


// معاملات Positional (ترتيبية)
void printInfo(String name, int age) {
  print('Name: $name, Age: $age');
}

// معاملات Named (مسماة)
void printInfoNamed({required String name, int age = 20}) {
  print('Name: $name, Age: $age');
}

void main() {
  printInfo('أحمد', 25); // Positional
  printInfoNamed(name: 'محمد'); // Named مع قيمة افتراضية
}

الدوال العليا (Higher-order)

الدوال العليا هي دوال تقبل دوال أخرى كمعاملات أو تُرجع دوالاً. إليك أهم الدوال العليا في Dart: 


void main() {
  var numbers = [1, 2, 3, 4, 5];
  
  // map: تحويل كل عنصر في القائمة
  var squared = numbers.map((n) => n * n).toList();
  print(squared); // [1, 4, 9, 16, 25]
  
  // where: تصفية العناصر
  var evens = numbers.where((n) => n % 2 == 0).toList();
  print(evens); // [2, 4]
  
  // reduce: دمج العناصر
  var sum = numbers.reduce((a, b) => a + b);
  print(sum); // 15
  
  // forEach: تنفيذ عملية لكل عنصر
  numbers.forEach((n) => print(n));
}

الدوال السهمية في Dart

ملاحظة هامة

الدوال السهمية هي صيغة مختصرة للدوال التي تحتوي على تعبير واحد فقط

الدالة العادية:


int square(int num) {
  return num * num;
}

الدالة السهمية:


int square(int num) => num * num;

متى نستخدم الدوال السهمية؟

  • عندما يكون جسم الدالة سطر واحد فقط
  • لتحسين قراءة الكود وتقليل التكرار
  • عند استخدام الدوال كمعاملات في دوال أخرى

نطاقات المتغيرات (Scoping) في Dart

ما هو نطاق المتغير؟

نطاق المتغير (Scope) هو المنطقة في الكود حيث يمكن الوصول إلى هذا المتغير واستخدامه. في Dart، نطاقات المتغيرات تحدد بواسطة الأقواس المعقوفة {}.

أنواع النطاقات في Dart

النطاق العام (Global Scope)


// متغير عام - يمكن الوصول إليه من أي مكان
var globalVar = 'أنا متغير عام';

void main() {
  print(globalVar); // يمكن الوصول إليه هنا
}
  • يُعرّف خارج أي دوال أو كلاسات
  • يمكن الوصول إليه من أي مكان في البرنامج
  • يجب استخدامه بحذر لتجنب التضارب

النطاق المحلي (Local Scope)


void main() {
  // متغير محلي - فقط داخل main()
  var localVar = 'أنا متغير محلي';
  print(localVar); // يعمل
}

// print(localVar); // خطأ - لا يمكن الوصول إليه هنا
  • يُعرّف داخل دوال أو كتل كود
  • يمكن الوصول إليه فقط داخل الكتلة التي عُرف فيها
  • يختفي بعد انتهاء تنفيذ الكتلة

نطاق الكتلة (Block Scope)


void main() {
  if (true) {
    // متغير كتلة - فقط داخل if
    var blockVar = 'أنا متغير كتلة';
    print(blockVar); // يعمل
  }
  
  // print(blockVar); // خطأ - لا يمكن الوصول إليه هنا
}
  • يُعرّف داخل أي كتلة محاطة بأقواس {}
  • يشمل حلقات for، if، while، وغيرها
  • أكثر تحديداً من النطاق المحلي

قواعد هامة في النطاقات

القاعدة الهرمية

الكود الداخلي يمكنه رؤية متغيرات الكود الخارجي، ولكن العكس غير صحيح

إخفاء المتغيرات

المتغير المحلي يخفي المتغير العام إذا تشابهت الأسماء

دورة الحياة

المتغيرات المحلية تُحذف بعد انتهاء تنفيذ الدالة أو الكتلة

الثوابت (final و const)

تتبع نفس قواعد النطاق ولكن لا يمكن تعديل قيمتها

مثال تطبيقي شامل 


var global = 'Global'; // نطاق عام

void main() {
  var mainVar = 'Main Function'; // نطاق محلي
  
  void innerFunction() {
    var innerVar = 'Inner Function'; // نطاق محلي آخر
    
    print(global); // متاح
    print(mainVar); // متاح
    print(innerVar); // متاح
    
    if (true) {
      var ifVar = 'If Block'; // نطاق كتلة
      print(ifVar); // متاح
    }
    
    // print(ifVar); // خطأ - غير متاح هنا
  }
  
  innerFunction();
  // print(innerVar); // خطأ - غير متاح هنا
}

// print(mainVar); // خطأ - غير متاح هنا

تحليل المثال

  • global متاح في كل مكان
  • mainVar متاح فقط داخل main() ودوالها الداخلية
  • innerVar متاح فقط داخل innerFunction()
  • ifVar متاح فقط داخل كتلة if

أفضل الممارسات

  • قلل استخدام المتغيرات العامة قدر الإمكان
  • استخدم أسماء مميزة للمتغيرات لتجنب الإخفاء
  • اجعل نطاق المتغير ضيقاً بقدر ما تحتاج

هياكل البيانات في Dart

List (قائمة)

القائمة (List) هي مجموعة مرتبة من العناصر يمكن أن تحتوي على تكرارات. 


void main() {
  // إنشاء قائمة
  List<String> names = ['أحمد', 'محمد', 'علي'];
  
  // إضافة عنصر
  names.add('خالد');
  
  // الوصول إلى عنصر
  print(names[0]); // أحمد
  
  // الطول
  print(names.length); // 4
  
  // طرق أخرى
  names.remove('محمد'); // حذف عنصر
  names.sort(); // ترتيب
  names.reversed; // عكس الترتيب
  names.contains('علي'); // true
}

Set (مجموعة)

المجموعة (Set) هي مجموعة غير مرتبة من العناصر الفريدة (بدون تكرار). 


void main() {
  // إنشاء مجموعة
  Set<int> numbers = {1, 2, 3, 3, 4};
  
  print(numbers); // {1, 2, 3, 4} (لا يوجد تكرار)
  
  // إضافة عنصر
  numbers.add(5);
  
  // حذف عنصر
  numbers.remove(2);
  
  // التحقق من وجود عنصر
  print(numbers.contains(3)); // true
  
  // العمليات على المجموعات
  var setA = {1, 2, 3};
  var setB = {3, 4, 5};
  
  print(setA.union(setB)); // {1, 2, 3, 4, 5} (اتحاد)
  print(setA.intersection(setB)); // {3} (تقاطع)
  print(setA.difference(setB)); // {1, 2} (فرق)
}

Map (خريطة)

الخريطة (Map) هي مجموعة من أزواج المفتاح-قيمة (Key-Value).


void main() {
  // إنشاء خريطة
  Map<String, int> ages = {
    'أحمد': 25,
    'محمد': 30,
    'خالد': 22
  };
  
  // الوصول إلى قيمة
  print(ages['أحمد']); // 25
  
  // إضافة أو تعديل
  ages['علي'] = 28;
  
  // حذف
  ages.remove('محمد');
  
  // التحقق من وجود مفتاح
  print(ages.containsKey('خالد')); // true
  
  // التكرار على الخريطة
  ages.forEach((name, age) {
    print('$name عمره $age سنة');
  });
}

مقدمة في البرمجة الكائنية (OOP)

الكلاسات والكائنات

البرمجة الكائنية (OOP) هي نموذج برمجة يعتمد على مفهوم الكائنات (Objects) التي تحتوي على بيانات ووظائف. 


// تعريف كلاس
class Person {
  // خصائص (Properties)
  String name;
  int age;
  
  // constructor (منشئ)
  Person(this.name, this.age);
  
  // method (وظيفة)
  void introduce() {
    print('أنا $name وعمري $age سنة');
  }
}

void main() {
  // إنشاء كائن
  var person = Person('أحمد', 25);
  
  // الوصول إلى الخصائص
  print(person.name); // أحمد
  
  // استدعاء الوظيفة
  person.introduce(); // أنا أحمد وعمري 25 سنة
}

نصيحة مهمة

البرمجة الكائنية تعتمد على أربع مبادئ أساسية: التغليف (Encapsulation)، الوراثة (Inheritance)، تعدد الأشكال (Polymorphism)، والتجريد (Abstraction). سنتعمق فيها في المحاضرات القادمة.

الكلاسات المدمجة في JavaScript

Date - التعامل مع التواريخ

// إنشاء كائن تاريخ للوقت الحالي
const today = new Date();

// الحصول على السنة الحالية
console.log(today.getFullYear()); // 2023 (السنة الحالية)

// إنشاء تاريخ محدد (28 يونيو 2023)
const eid = new Date(2023, 5, 28); 
// ملاحظة: الشهر يبدأ من 0 (0 = يناير، 5 = يونيو)

// الحصول على يوم الأسبوع (0 = الأحد، 1 = الإثنين، ...)
console.log(eid.getDay()); // 3 (يوم الأربعاء)
أهم دوال الكلاس Date:
  • .getDate() - يوم الشهر (1-31)
  • .getMonth() - الشهر (0-11)
  • .getFullYear() - السنة (4 أرقام)
  • .getDay() - يوم الأسبوع (0-6)
  • .getHours() - الساعات (0-23)
  • .toLocaleDateString() - تاريخ بصيغة محلية
