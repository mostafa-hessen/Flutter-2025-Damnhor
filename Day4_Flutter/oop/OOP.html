ليه نستخدم الـ OOP أصلاً؟

تخيل معايا عندك أكتر من عربية لعب، وكل عربية ليها:

  • لون
  • نوع

  • وزرار يشغلها

لو كتبت كود عادي لكل عربية لوحدها، الدنيا هتبقى زحمة ومكركبة ومليانة تكرار ومش منظمة.

المشاكل من غير OOP

كود عادي (بدون OOP)

// بيانات عربية 1
String car1Name = "BMW";
String car1Color = "أحمر";
int car1Speed = 100;

void driveCar1() {
  print("السيارة $car1Name تمشي بسرعة $car1Speed");
}

// بيانات عربية 2
String car2Name = "Kia";
String car2Color = "أزرق";
int car2Speed = 80;

void driveCar2() {
  print("السيارة $car2Name تمشي بسرعة $car2Speed");
}

العيوب:

  • ✗ تكرار كبير جدًا للكود
  • ✗ صعب تضيف ميزة جديدة لكل العربيات
  • ✗ لو غيرت حاجة، لازم تعدل في كل مكان

كود باستخدام OOP

class Car {
  String name;
  String color;
  int speed;

  // Constructor: بيدينا القيم لما نعمل عربية
  Car(this.name, this.color, this.speed);

  // دالة بتخلي العربية تمشي
  void drive() {
    print("السيارة $name لونها $color تمشي بسرعة $speed");
  }
}

void main() {
  Car car1 = Car("BMW", "أحمر", 100);
  Car car2 = Car("Kia", "أزرق", 80);

  car1.drive();
  car2.drive();
}

المميزات:

  • ✓ لا يوجد تكرار للكود
  • ✓ إضافة ميزات جديدة سهلة
  • ✓ التعديل في مكان واحد فقط

  void deposit() { ... } // عام

🧱 مفاهيم OOP الأساسية

Class & Object

الكلاس هو القالب الذي يحدد الخصائص والوظائف، والكائن هو النسخة الفعلية منه. 

class Car { ... }
Car bmw = Car();

Encapsulation

إخفاء البيانات الداخلية وتوفير واجهة للتعامل مع الكائن. 

class Wallet {
  double _balance;
  
  void deposit(double amount) {
    if (amount > 0) _balance += amount;
  }
}

Inheritance

قدرة الكلاس على وراثة خصائص ووظائف كلاس آخر. 

class Animal { ... }
class Dog extends Animal { ... }

Polymorphism

قدرة الكائنات على اتخاذ أشكال متعددة واستجابة مختلفة لنفس الرسالة. 

class Animal {
  void speak() => print("...");
}

class Dog extends Animal {
  @override
  void speak() => print("هاو هاو");
}

Abstraction

إظهار الوظائف الأساسية وإخفاء تفاصيل التنفيذ. 

abstract class Shape {
  void draw(); // دالة مجردة
}

class Circle extends Shape {
  @override
  void draw() { ... }
}

Static

متغيرات أو دوال مشتركة بين كل الكائنات من نفس الكلاس. 

class MathUtils {
  static const double pi = 3.14;
  
  static double square(double n) => n * n;
}

// الاستخدام
double area = MathUtils.pi * MathUtils.square(5);

🔐

التغليف (Encapsulation)

🎈 تخيل إن معاك "محفظة إلكترونية" زي Vodafone Cash:

  • فيها فلوس (balance)
  • تقدر تحط فلوس (deposit)
  • تقدر تسحب (withdraw)
  • بس ماينفعش أي حد يغير الرصيد من بره الكلاس!

✨ يعني إيه Encapsulation؟

يعني نغلف البيانات ونحميها، ونخلي الناس تتعامل معاها من بوابات معينة (getters & setters) بس.

🏦 مثال بسيط على محفظة إلكترونية:

class Wallet {
  String _owner; // خاص - برا الكلاس مش شايفه
  double _balance = 0; // خاص - محدش يقدر يعدله مباشرة

  Wallet(this._owner);

  // Getter لعرض الرصيد
  double get balance => _balance;

  // إيداع
  void deposit(double amount) {
    if (amount > 0) {
      _balance += amount;
      print("تم إضافة $amount جنيه. الرصيد الحالي: $_balance");
    } else {
      print("المبلغ غير صحيح");
    }
  }

  // سحب
  void withdraw(double amount) {
    if (amount <= _balance) {
      _balance -= amount;
      print("تم سحب $amount جنيه. الرصيد الحالي: $_balance");
    } else {
      print("رصيد غير كافي");
    }
  }
}

void main() {
  Wallet myWallet = Wallet("مصطفى");

  myWallet.deposit(200);
  myWallet.withdraw(50);
  print("الرصيد النهائي: ${myWallet.balance}");

  // محاولة لتعديل الرصيد من بره (مش هتشتغل)
  // myWallet._balance = 100000; ❌
}

class ATM {
  double _balance = 1000;

  double get balance => _balance;

  void deposit(double amount) {
    if (amount > 0) {
      _balance += amount;
      print("تم إيداع $amount. الرصيد الحالي: $_balance");
    } else {
      print("قيمة الإيداع غير صحيحة");
    }
  }

  void withdraw(double amount) {
    if (amount <= _balance) {
      _balance -= amount;
      print("تم سحب $amount. الرصيد الحالي: $_balance");
    } else {
      print("رصيد غير كافي");
    }
  }

  void checkBalance() {
    print("رصيدك الحالي: $_balance");
  }
}

void main() {
  ATM atm = ATM();
  atm.checkBalance();
  atm.deposit(500);
  atm.withdraw(200);
  atm.checkBalance();
}

الوراثة (Inheritance)

التعريف

الوراثة معناها إن كلاس (فئة) جديدة تقدر ترث الخصائص والدوال من كلاس موجود أصلاً، وبكده تقدر تستخدم وتعدل أو تضيف عليها بدون ما تعيد كتابة نفس الكود.

مثال واقعي

فكر معايا في حياة الطلبة:

  • عندنا فئة عامة اسمها "موظف" (Employee).
  • عندنا فئات فرعية زي "طالب دوام جزئي" و "طالب دوام كامل" اللي هما أنواع موظفين، كل واحد ليه وظائف وخصائص معينة.
  • كل هؤلاء الموظفين يشتركوا في خصائص عامة (اسم، رقم تعريف، راتب)، لكن في نفس الوقت عندهم تصرفات مختلفة.

مثال برمجي (Dart)

class Employee {
  String name;
  double salary;

  Employee(this.name, this.salary);

  void work() {
    print("$name يعمل براتب $salary");
  }
}

class PartTimeStudent extends Employee {
  int hours;

  PartTimeStudent(String name, double salary, this.hours) : super(name, salary);

  void study() {
    print("$name يدرس $hours ساعات في الأسبوع");
  }

  @override
  void work() {
    print("$name يعمل دوام جزئي براتب $salary");
  }
}

void main() {
  PartTimeStudent ali = PartTimeStudent("علي", 5000, 20);
  ali.work();   // علي يعمل دوام جزئي براتب 5000
  ali.study();  // علي يدرس 20 ساعات في الأسبوع
}

الفائدة العملية

في شغلك أو مشروعك، بدل ما تكتب كود مكرر لكل نوع موظف أو كائن مختلف، تستخدم الوراثة عشان تعيد استخدام الكود وتنظمه.

✨

Abstract Class مع Extends (الكلاس المجرد + الوراثة)

التعريف

الكلاس المجرد هو قالب لا يمكن إنشاء كائن منه مباشرة، لكنه يحدد دوال يجب أن تُنفذ في الكلاسات اللي تورثه.

مثال واقعي

تخيل عندك تطبيق بنظام توصيل طلبات:

  • عندك كلاس مجرد اسمه DeliveryPerson يعرف وظيفة أساسية وهي التوصيل deliver().
  • عندك كلاس فرعي مثل BikeCourier و CarCourier كل واحد منهم له طريقة مختلفة للتوصيل.

مثال برمجي (Dart)

abstract class DeliveryPerson {
  String name;

  DeliveryPerson(this.name);

  void deliver();  // دالة مجردة لازم تنفذ في الكلاسات الفرعية
}

class BikeCourier extends DeliveryPerson {
  BikeCourier(String name) : super(name);

  @override
  void deliver() {
    print("$name يوصل الطلب بالدراجة");
  }
}

class CarCourier extends DeliveryPerson {
  CarCourier(String name) : super(name);

  @override
  void deliver() {
    print("$name يوصل الطلب بالسيارة");
  }
}

void main() {
  List<DeliveryPerson> couriers = [BikeCourier("أحمد"), CarCourier("محمود")];
  
  for (var courier in couriers) {
    courier.deliver();
  }
}

الفائدة العملية

تقدر تنشئ نظام معياري، تعرف فيه الوظائف الأساسية وتترك التفاصيل للكلاسات المتخصصة، وهذا يسهل صيانة الكود وتطويره.

📜

Interface (الواجهة)

التعريف

الواجهة هي عقد يحدد مجموعة من الدوال التي يجب على الكلاس الذي ينفذها (implements) أن يقدمها. في Dart، الواجهات تتحقق باستخدام abstract class أو class مع implements.

مثال واقعي

تخيل نظام للدفع في متجر إلكتروني:

  • عندنا واجهة PaymentMethod تحتوي على دالة pay(amount)
  • هناك عدة طرق دفع: بطاقة ائتمان، PayPal، ومحفظة إلكترونية. كل طريقة تنفيذ مختلف.

مثال برمجي (Dart)

abstract class PaymentMethod {
  void pay(double amount);
}

class CreditCard implements PaymentMethod {
  @override
  void pay(double amount) {
    print("تم الدفع $amount باستخدام بطاقة الائتمان");
  }
}

class PayPal implements PaymentMethod {
  @override
  void pay(double amount) {
    print("تم الدفع $amount باستخدام PayPal");
  }
}

void main() {
  PaymentMethod method = CreditCard();
  method.pay(150.0);  // تم الدفع 150 باستخدام بطاقة الائتمان

  method = PayPal();
  method.pay(200.0);  // تم الدفع 200 باستخدام PayPal
}

الفائدة العملية

لو انت بتبني نظام فيه أكثر من طريقة لتنفيذ وظيفة واحدة (مثل الدفع، الإشعارات، أو تسجيل الدخول)، بتستخدم الواجهات لتوحيد الكود مع السماح بتنفيذات مختلفة.

⚡

Static (المتغيرات والدوال الثابتة)

التعريف

المتغير أو الدالة الثابتة هي شيء مشترك بين كل الكائنات ولا يحتاج إنشاء كائن للوصول إليه. تستخدم لحفظ بيانات أو وظائف عامة.

مثال واقعي

في تطبيق Flutter: عندك ثيم ثابت لكل التطبيق أو اعدادات عامة:

  • ثابت لتحديد الهوامش (padding)
  • ثابت لعنوان URL لواجهة برمجة التطبيقات

مثال Flutter عملي

class AppSettings {
  static const double defaultPadding = 16.0;
  static const String apiEndpoint = "https://api.myapp.com";
  
  static void printSettings() {
    print("Padding: $defaultPadding, API URL: $apiEndpoint");
  }
}

void main() {
  print(AppSettings.defaultPadding);   // 16.0
  AppSettings.printSettings();          // Padding: 16.0, API URL: https://api.myapp.com
}

الفائدة العملية

لو عندك قيمة أو وظيفة مش بتتغير أو تخص التطبيق كله مش كائن معين، تستخدم static علشان توفر الموارد وتسهّل الاستخدام.

1 Class و Object

مثال حقيقي: نموذج بيانات (Data Model) للـUser في التطبيق

class User {
  String name;
  int age;

  User(this.name, this.age);
}

void main() {
  User user1 = User('أحمد', 25);  // object من class User

}

استخدام عملي:

عادة كل بيانات تُجلب من API أو قاعدة بيانات تتحول لأوبجكت لتسهيل التعامل معها في واجهة المستخدم (مثلاً: عرض اسم المستخدم في الصفحة الرئيسية).

2 Encapsulation (التغليف)

مثال حقيقي: حماية بيانات داخل كلاس وإخفاؤها

class BankAccount {
  double _balance = 0; // خاص

  double get balance => _balance;  // طريقة قراءة فقط

  void deposit(double amount) {
    if (amount > 0) _balance += amount;
  }

  void withdraw(double amount) {
    if (amount > 0 && amount <= _balance) _balance -= amount;
  }
}

void main() {
  BankAccount acc = BankAccount();
  acc.deposit(100);
  acc.withdraw(50);
  print(acc.balance);  // 50
}

استخدام عملي:

في التطبيقات المالية أو المحفظة الإلكترونية، نحمي بيانات الرصيد ولا نسمح بالتعديل المباشر، فقط من خلال دوال محددة.

3 Inheritance و super

مثال حقيقي: بناء Widget مخصص يرث من StatelessWidget

import 'package:flutter/material.dart';

class MyButton extends StatelessWidget {
  final String text;
  final VoidCallback onPressed;

  MyButton(this.text, this.onPressed);

  @override
  Widget build(BuildContext context) {
    return ElevatedButton(onPressed: onPressed, child: Text(text));
  }
}

class RedButton extends MyButton {
  RedButton(String text, VoidCallback onPressed)
      : super(text, onPressed);  // استدعاء constructor الأب

  @override
  Widget build(BuildContext context) {
    return ElevatedButton(
      onPressed: onPressed,
      style: ElevatedButton.styleFrom(primary: Colors.red),
      child: Text(text),
    );
  }
}

استخدام عملي:

تخصيص Widgets عن طريق الوراثة لتغيير الشكل أو الوظائف مع إعادة استخدام الكود.

4 Polymorphism (تعدد الأشكال)

مثال حقيقي: Widgets تستخدم polymorphism عند التعامل مع قائمة متنوعة

abstract class Animal {
  void speak();
}

class Dog implements Animal {
  @override
  void speak() => print("هاو هاو");
}

class Cat implements Animal {
  @override
  void speak() => print("مياو");
}

void main() {
  List animals = [Dog(), Cat()];
  animals.forEach((animal) => animal.speak());
}

استخدام عملي:

في Flutter، كل Widget يرث من Widget لكن كل نوع منهم يعرض ويعمل بشكل مختلف. تستطيع التعامل معهم جميعًا عبر نوع مشترك.

5 Abstract مع Extends

مثال حقيقي: إنشاء قاعدة لشاشات التطبيق

import 'package:flutter/material.dart';

abstract class BasePage extends StatelessWidget {
  Widget buildContent();

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('تطبيق')),
      body: buildContent(),
    );
  }
}

class HomePage extends BasePage {
  @override
  Widget buildContent() {
    return Center(child: Text('الصفحة الرئيسية'));
  }
}

استخدام عملي:

لإنشاء هيكل موحد لجميع صفحات التطبيق مع إمكانية تخصيص المحتوى.

8 Enum

مثال حقيقي: حالة تحميل البيانات في التطبيق

enum LoadingStatus { loading, completed, error }

void main() {
  LoadingStatus status = LoadingStatus.loading;

  switch (status) {
    case LoadingStatus.loading:
      print("جاري التحميل...");
      break;
    case LoadingStatus.completed:
      print("تم التحميل!");
      break;
    case LoadingStatus.error:
      print("حدث خطأ");
      break;
  }
}
