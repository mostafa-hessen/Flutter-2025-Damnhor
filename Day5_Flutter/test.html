مقارنة الكود

procedural.dart


// بيانات متفرقة
List<String> brands = [];
List<String> models = [];
List<int> years = [];
List<int> speeds = [];

// إضافة سيارة
void addCar(String brand, String model, int year) {
  brands.add(brand);
  models.add(model);
  years.add(year);
  speeds.add(0);
}

// عرض التفاصيل
void display(int index) {
  print('Brand: ${brands[index]}');
  print('Model: ${models[index]}');
  print('Year: ${years[index]}');
  print('Speed: ${speeds[index]} km/h');
}
مشاكل: بيانات متفرقة - صعوبة التوسع - محدودية إعادة الاستخدام

oop.dart


// تعريف الكلاس
class Car {
  String brand;
  String model;
  int year;
  int speed = 0;

  Car(this.brand, this.model, this.year);

  // تسريع السيارة
  void accelerate(int amount) {
    speed += amount;
  }

  // عرض التفاصيل
  void display() {
    print('Brand: $brand');
    print('Model: $model');
    print('Year: $year');
    print('Speed: $speed km/h');
  }
}

💡 مثال عملي من مصنع السيارات

لنفترض أننا نريد إضافة نوع خاص من السيارات (كهربائية) مع خاصية إضافية (مستوى البطارية):

في البرمجة الإجرائية:

- نضيف List جديدة للبطارية
- نعدل جميع الدوال لتأخذ في الاعتبار هذه الخاصية الجديدة

في البرمجة الكائنية:

- ننشئ كلاس جديد ElectricCar يرث من Car
- نضيف الخاصية الجديدة والسلوك المطلوب فقط


class ElectricCar extends Car {
  double batteryLevel = 100;

  ElectricCar(String brand, String model, int year)
      : super(brand, model, year);

  void charge(double amount) {
    batteryLevel = batteryLevel + amount > 100 
        ? 100 
        : batteryLevel + amount;
  }

  void display() {
    super.display();
    print('Battery: $batteryLevel%');
  }
}
