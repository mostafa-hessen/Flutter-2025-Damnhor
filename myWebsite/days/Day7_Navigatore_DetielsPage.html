مقدمة عن التنقل بين الصفحات

في تطبيقات Flutter، يعتبر نظام التنقل بين الصفحات من الأساسيات المهمة. هناك طريقتان رئيسيتان للتنقل:

directions 1. التنقل باستخدام MaterialPageRoute

الطريقة الأساسية للتنقل، حيث يتم إنشاء الصفحة الجديدة مباشرة عند التنقل.

  • مناسبة للتطبيقات الصغيرة
  • سهلة الفهم والتطبيق
  • لا تتطلب إعداد مسبق

alt_route 2. التنقل باستخدام Named Routes

طريقة أكثر تنظيماً، حيث يتم تعريف المسارات مسبقاً ثم استدعاؤها عند الحاجة.

  • مناسبة للتطبيقات الكبيرة
  • أسهل في الصيانة والتعديل
  • تتطلب إعداداً مسبقاً
directions

الطريقة الأولى: MaterialPageRoute

هذه الطريقة تعتمد على إنشاء الصفحة الجديدة مباشرة عند التنقل إليها باستخدام MaterialPageRoute.

code الكود الكامل 

import 'package:flutter/material.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Flutter Demo',
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: HomePage(),
    );
  }
}

class HomePage extends StatelessWidget {
  final List<Item> items = [
    Item(id: 1, title: 'عنصر 1', description: 'وصف العنصر الأول'),
    Item(id: 2, title: 'عنصر 2', description: 'وصف العنصر الثاني'),
    Item(id: 3, title: 'عنصر 3', description: 'وصف العنصر الثالث'),
  ];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('القائمة الرئيسية')),
      body: ListView.builder(
        itemCount: items.length,
        itemBuilder: (context, index) {
          return ItemCard(
            item: items[index],
            onTap: () {
              Navigator.push(
                context,
                MaterialPageRoute(
                  builder: (context) => DetailsPage(item: items[index]),
                ),
              );
            },
          );
        },
      ),
    );
  }
}

class Item {
  final int id;
  final String title;
  final String description;

  Item({required this.id, required this.title, required this.description});
}

class ItemCard extends StatelessWidget {
  final Item item;
  final VoidCallback onTap;

  const ItemCard({
    Key? key,
    required this.item,
    required this.onTap,
  }) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Card(
      margin: EdgeInsets.all(8),
      child: InkWell(
        onTap: onTap,
        child: Padding(
          padding: EdgeInsets.all(16),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Text(
                item.title,
                style: Theme.of(context).textTheme.headline6,
              ),
              SizedBox(height: 8),
              Text(
                item.description,
                style: Theme.of(context).textTheme.bodyText2,
                maxLines: 2,
                overflow: TextOverflow.ellipsis,
              ),
            ],
          ),
        ),
      ),
    );
  }
}

class DetailsPage extends StatelessWidget {
  final Item item;

  const DetailsPage({
    Key? key,
    required this.item,
  }) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text(item.title),
      ),
      body: Padding(
        padding: EdgeInsets.all(16),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Text(
              item.title,
              style: Theme.of(context).textTheme.headline4,
            ),
            SizedBox(height: 16),
            Text(
              item.description,
              style: Theme.of(context).textTheme.bodyText2,
            ),
          ],
        ),
      ),
    );
  }
}

عند النقر على العنصر، يتم إنشاء صفحة DetailsPage جديدة وتمرير العنصر المحدد إليها.

تعرض تفاصيل العنصر الذي تم النقر عليه، مع إمكانية العودة للصفحة السابقة باستخدام زر الرجوع في AppBar.

الطريقة الثانية: Named Routes

هذه الطريقة تعتمد على تعريف مسارات مسبقة ثم استدعاؤها عند الحاجة، مما يجعل الكود أكثر تنظيماً وسهولة في الصيانة.

code الكود الكامل 

import 'package:flutter/material.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Flutter Demo',
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      // تعريف المسارات
      routes: {
        '/': (context) => HomePage(),
        '/details': (context) {
          final item = ModalRoute.of(context)!.settings.arguments as Item;
          return DetailsPage(item: item);
        },
      },
      initialRoute: '/',
    );
  }
}

class HomePage extends StatelessWidget {
  final List<Item> items = [
    Item(id: 1, title: 'عنصر 1', description: 'وصف العنصر الأول'),

    Item(id: 3, title: 'عنصر 3', description: 'وصف العنصر الثالث'),

مقارنة بين الطريقتين

المعيار MaterialPageRoute Named Routes
التعقيد بسيطة وسهلة الفهم أكثر تعقيداً وتتطلب إعداداً مسبقاً
تمرير البيانات سهل ومباشر عبر constructor يتم عبر arguments ويحتاج لتحويل النوع
حجم التطبيق مناسبة للتطبيقات الصغيرة مناسبة للتطبيقات الكبيرة
الصيانة قد تصبح معقدة مع زيادة الصفحات سهلة الصيانة والتعديل
إعادة الاستخدام لا توجد إدارة مركزية للمسارات إمكانية إعادة استخدام المسارات

lightbulb نصائح لاختيار الطريقة المناسبة

  • استخدم MaterialPageRoute للتطبيقات الصغيرة أو النماذج الأولية
  • استخدم Named Routes للتطبيقات الكبيرة أو التي تحتوي على العديد من الصفحات
  • يمكنك الجمع بين الطريقتين في تطبيق واحد حسب الحاجة
  • للتطبيقات المعقدة، يمكن استخدام حزم مثل go_router أو fluro
